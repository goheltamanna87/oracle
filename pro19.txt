declare
  r_emp emp%rowtype;
  n_emp_id emp.empno%type:=101;
begin
   select*into r_emp from emp where empno=n_emp_id;
dbms_output.put_line(r_emp.ename);
end;
/
         or
declare
r_emp emp%rowtype;
begin
   select*into r_emp from emp where empno=101;
dbms_output.put_line(r_emp.empno);
dbms_output.put_line(r_emp.ename);
end;
/
